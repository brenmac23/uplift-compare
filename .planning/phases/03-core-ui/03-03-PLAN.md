export function findSection(result: ScoringResult, sectionId: string | null): SectionResult | null {
  if (!sectionId) return null;
  return result.sections.find(s => s.id === sectionId) ?? null;
}

**Live Scoring (the core UX):**
- Copy project.inputs into local state via useState on mount
- Use useMemo to compute existingResult = scoreExisting(inputs) and proposedResult = scoreProposed(inputs) whenever inputs change
