  // Financial — qnzpe in whole NZD dollars

  // Production Activity
  hasStudioLease: boolean;             // existing B1: 2pts (existing only)
  hasPreviousQNZPE: boolean;           // existing B2 / proposed A1: 2pts
  hasAssociatedContent: boolean;       // existing B3: 1pt / proposed A2: 2pts
  shootingNZPercent: number;           // existing B4 / proposed A3: tiered
  regionalPercent: number;             // existing B5 / proposed A4: tiered (different thresholds)
  picturePostPercent: number;          // existing B6 / proposed A5: tiered
  soundPostPercent: number;            // existing B7 / proposed A6: tiered
  vfxPercent: number;                  // existing B8 / proposed A7: tiered
  conceptPhysicalPercent: number;      // existing B9 / proposed A8: tiered

  // Personnel — percentages
  castPercent: number;                 // existing C1 / proposed B1: tiered
  crewPercent: number;                 // existing C2 / proposed B2: tiered
  maoriCrewPercent: number;            // existing C3: 1pt (existing only)

  // Personnel — counts
  atlCount: number;                    // 0-3, existing C4 / proposed B3
  btlKeyCount: number;                 // existing C5 / proposed B4: counted, cap 4
  btlAdditionalCount: number;          // existing C6 / proposed B5: counted (1pt vs 0.5pt each)
  hasLeadCast: boolean;                // existing C7 / proposed B6: binary
  supportingCastCount: number;         // 0-3, existing C8 / proposed B7
  castingLevel: 'none' | 'associate' | 'director';  // existing C9 / proposed B8
  hasLeadCastMaori: boolean;           // existing C10: 2pts (existing only)

  // Skills & Development
  hasMasterclass: boolean;             // existing D1: 2pts (existing only)
  hasIndustrySeminars: boolean;        // proposed C1: 1pt
  hasEdSeminars: boolean;              // existing D2 / proposed C2: 1pt
  attachmentCount: number;             // existing D3 / proposed C3: tiered by QNZPE
  internshipCount: number;             // existing D4 / proposed C4: tiered by QNZPE

  // Innovation (existing only)
  hasKnowledgeTransfer: boolean;       // existing E1: 2pts
  commercialAgreementPercent: number;  // existing E2: tiered (0.25/0.5/1% of QNZPE)
  infrastructureInvestment: number;    // existing E3: tiered ($500k/$1m/$2m) in whole NZD

  // Marketing
  premiereType: 'none' | 'nz' | 'world';  // existing F1: selector (existing only shape)

  hasTourismMarketing: boolean;        // existing F3: 3pts (existing only — replaced by D3+D4 in proposed)
  hasTourismPartnership: boolean;      // existing F4 / proposed D4: 3pts existing / 4pts proposed
  hasLocationAnnouncement: boolean;    // proposed D3: 1pt (proposed only)
}
